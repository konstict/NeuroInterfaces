vs_BuildTools.exe - Разработка классических приложений для Windows
cmake latest version
dlib latest version (or 20.0)
for RTX 3050: cuda_12.4.0_551.61, cudnn_9.4.0 (other mb too)


cd %USERPROFILE%/downloads
git clone https://github.com/davisking/dlib.git (or download from github as zip)
cd dlib
git checkout v20.0 (optional)

mkdir build
cd build

cmake .. -G "Visual Studio 17 2022" -A x64 ^
 -DDLIB_USE_CUDA=1 ^
 -DUSE_AVX_INSTRUCTIONS=1 ^
 -DDLIB_NO_GUI_SUPPORT=1
cmake --build . --config Release

python -m pip install --upgrade pip setuptools wheel
python setup.py bdist_wheel

cd dist
(and you have dlib-20.0.0-cp311-cp311-win_amd64.whl, then you need to pip install this file by pip install)
